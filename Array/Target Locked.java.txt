import java.util.Scanner;
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int arr[] = new int[n];
        for(int i =0;i<n;i++){
            arr[i] = sc.nextInt();
        }
        int key = sc.nextInt();
        boolean x = false;
        for(int i =0;i<n;i++)
        {
            if(arr[i] == key)
            {
                System.out.println(i);
                x = true;
                break;
            }
        
        }
        if(x == false)
            System.out.print("-1");
    }
}
