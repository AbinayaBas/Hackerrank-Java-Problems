import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();         
        int square = n * n;           
        int sum = 0;
        while (square > 0) {
            sum += square % 10;
            square = square / 10;
        }
        if (sum == n) {
            System.out.println("true");
        } else {
            System.out.println("false");
        }
    }
}
